88 function rush_op_set_drupal_variables($params) {
89 
90  if (empty($params['drupal_variables'])) {
91  return rush_empty_error('$params[\'drupal_variables\']', __FUNCTION__);
92  }
93 
94  if (empty($params['build']['site_alias_record'])) {
95  return rush_empty_error('$params[\'build\'][\'site_alias_record\']', __FUNCTION__);
96  }
97 
98  // These two variables will be passed to rush_invoke inside of the foreach
99  // loop below. These two variables do not change based on the loop.
100  $alias = $params['build']['site_alias_record'];
101  // Use the exact flag because the parameters file should only use exact
102  // variable names.
103  $options = array(
104  '--exact',
105  );
106 
107  foreach ($params['drupal_variables'] as $variable_name => $variable_value) {
108  // The variable name and value are passed to "drush variable-set"
109  $args = array(
110  $variable_name,
111  $variable_value,
112  );
113 
114  // If a variable-set attempt fails (rush_invoke returns TRUE or FALSE)
115  // then return FALSE immediately, even if the foreach looping has not
116  // reached it's last item.
117  if (FALSE === rush_invoke($alias, 'variable-set', $args, $options, __FUNCTION__)) {
118  return FALSE;
119  }
120  }
121 
122  // If the foreach loop was completed without any fails, return TRUE because
123  // all variable-sets have succeeded.
124  return TRUE;
125 }
