81 function rush_op_create_build_directory($params) {
82 
83  if (empty($params['build']['directory_path'])) {
84  return rush_empty_error('$params[\'build\'][\'directory_path\']', __FUNCTION__);
85  }
86 
87  if (empty($params['build']['backups_directory_path'])) {
88  return rush_empty_error('$params[\'build\'][\'backups_directory_path\']', __FUNCTION__);
89  }
90 
91  if (empty($params['build']['key'])) {
92  return rush_empty_error('$params[\'build\'][\'key\']', __FUNCTION__);
93  }
94 
95  if (empty($params['build']['directory_name'])) {
96  return rush_empty_error('$params[\'build\'][\'directory_name\']', __FUNCTION__);
97  }
98 
99  $build_directory_path = $params['build']['directory_path'];
100  // If no previous build directory exists, create one.
101  if (!is_dir($build_directory_path)) {
102  drush_mkdir($build_directory_path, TRUE);
103  // Pause for a second to get directory made before other operations need it.
104  sleep(1);
105  }
106  else {
107  // A previous directory does exist, alert the user.
108  $msg = dt('[NOTICE]: *** Directory !path already exists. Consider adding the operation `destroy_build_directory` to your rush file. ***', array(
109  '!path' => $build_directory_path,
110  ));
111  drush_log($msg, 'notice');
112 
113  // Strip the directory forward slash off to prepare the rename action.
114  $old_dir = rtrim($build_directory_path, '/');
115  $new_dir = $params['build']['backups_directory_path'] . $params['build']['directory_name'] . '.' . date("Y-m-d-H-i-s") . '.' . $params['build']['key'];
116 
117  if (!is_dir($params['build']['backups_directory_path'])) {
118  drush_mkdir($params['build']['backups_directory_path'], TRUE);
119  }
120 
121  if (!rename($old_dir, $new_dir)) {
122  // If the directory could not be successfully renamed then return error.
123  return drush_set_error('RUSH_FILESYSTEM_OP_ERROR', dt('[ERROR]: *** Unable to rename directory !path ***', array('!path' => $build_directory_path)));
124  }
125  else {
126  $msg = dt('[RENAME DIRECTORY]: !path to !new_dir.', array(
127  '!path' => $build_directory_path,
128  '!new_dir' => $new_dir,
129  ));
130  drush_log($msg, 'info');
131 
132  }
133  // Now create the directory.
134  drush_mkdir($build_directory_path, TRUE);
135  // Pause for a second to get directory made before other operations need it.
136  sleep(1);
137  }
138 
139  // Error checking. Confirm the new directory exists.
140  if (!is_dir($build_directory_path)) {
141  return drush_set_error('RUSH_FILESYSTEM_OP_ERROR', dt('[ERROR]: *** Unable to create directory !path ***', array('!path' => $build_directory_path)));
142  }
143  else {
144  $msg = dt('[CREATE DIRECTORY]: !path.', array('!path' => $build_directory_path));
145  drush_log($msg, 'success');
146 
147  return TRUE;
148 
149  }
150 }
