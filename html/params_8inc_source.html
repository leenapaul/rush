18 /**
19  * Parse rush parameter files and return an array of parameters.
20  *
21  * The following is the order of parsing:
22  * 1. Get params from the global params ini file if it exists.
23  * 2. Get params from the global params php file if it exists.
24  * - Params in the php file will override the ini file.
25  * 3. Get params from the job params ini file if it exists.
26  * 4. Merge global params with job params, letting job params override global.
27  * 5. Get params from the job params php file if it exists.
28  * - Params in the job php file will override all params to this point.
29  *
30  * @see doc-parameters-list
31  * @return array
32  * An associative array of all rush parameters.
33  */
34 function rush_get_params() {
35 
36  $global_locations = rush_get_global_locations();
37  $params = array();
38  // IMPORT GLOBAL PARAMETERS
39  // Get params from the global params ini file if it exists.
40  $params_ini_file = $global_locations['RUSH_ENVIRONMENT_DIRECTORY_PATH'] . 'params.ini';
41  if (file_exists($params_ini_file)) {
42  $msg = $msg = dt('[LOAD PARAMS FILE]: !file.', array('!file' => $params_ini_file));
43  drush_log($msg, 'notice');
44  $params = rush_parse_ini($params_ini_file);
45  }
46  // Get params from the global params php file if it exists.
47  // Params in the php file will override the ini file.
48  $params_php_file = $global_locations['RUSH_ENVIRONMENT_DIRECTORY_PATH'] . 'params.php';
49  if (file_exists($params_php_file)) {
50  $msg = $msg = dt('[LOAD PARAMS FILE]: !file.', array('!file' => $params_php_file));
51  drush_log($msg, 'notice');
52 
53  include_once "$params_php_file";
54  }
55  unset($params_ini_file, $params_php_file);
56  // IMPORT TOP JOB PARAMETERS IF THERE IS A TOP JOB
57  // If the job called is a sub job, then pull parameters from the top
58  // job first.
59  if (!empty($global_locations['RUSH_JOB_TOP_DIRECTORY_PATH'])) {
60  // Get params from the top job params ini file if it exists.
61  $params_ini_file = $global_locations['RUSH_JOB_TOP_DIRECTORY_PATH'] . 'params.ini';
62  if (file_exists($params_ini_file)) {
63  $msg = $msg = dt('[LOAD PARAMS FILE]: !file.', array('!file' => $params_ini_file));
64  drush_log($msg, 'notice');
65  $top_job_params = rush_parse_ini($params_ini_file);
66  if (!empty($top_job_params)) {
67  // Merge top job params with params set to this point, letting top job
68  // params override.
69  $params = _rush_array_merge_recursive_distinct($params, $top_job_params);
70  }
71  }
72  // Get params from the top job params php file if it exists.
73  // Params in the php file will override all params to this point.
74  $params_php_file = $global_locations['RUSH_JOB_TOP_DIRECTORY_PATH'] . 'params.php';
75  if (file_exists($params_php_file)) {
76  $msg = $msg = dt('[LOAD PARAMS FILE]: !file.', array('!file' => $params_php_file));
77  drush_log($msg, 'notice');
78 
79  include_once "$params_php_file";
80  }
81  }
82  unset($params_ini_file, $params_php_file);
83  // IMPORT JOB PARAMETERS
84  // Get params from the job params ini file if it exists.
85  $params_ini_file = $global_locations['RUSH_JOB_DIRECTORY_PATH'] . 'params.ini';
86  if (file_exists($params_ini_file)) {
87  $msg = $msg = dt('[LOAD PARAMS FILE]: !file.', array('!file' => $params_ini_file));
88  drush_log($msg, 'notice');
89  $job_params = rush_parse_ini($params_ini_file);
90  // Merge job params with params set to this point, letting job params
91  // override.
92  if (!empty($job_params)) {
93  $params = _rush_array_merge_recursive_distinct($params, $job_params);
94  }
95  }
96  // Get params from the job params php file if it exists.
97  // Params in the php file will override all params to this point.
98  $params_php_file = $global_locations['RUSH_JOB_DIRECTORY_PATH'] . 'params.php';
99  if (file_exists($params_php_file)) {
100  $msg = $msg = dt('[LOAD PARAMS FILE]: !file.', array('!file' => $params_php_file));
101  drush_log($msg, 'notice');
102 
103  include_once "$params_php_file";
104  }
105  $command_line = array();
106  // Add in options passed on the command line.
107  if (drush_get_option('domain', FALSE)) {
108  $command_line['domain'] = $params['build']['domain'] = drush_get_option('domain');
109  }
110  if (drush_get_option('branch', FALSE)) {
111  $command_line['branch'] = $params['repo']['branch'] = drush_get_option('branch');
112  }
113  if (drush_get_option('new-branch', FALSE)) {
114  $command_line['new-branch'] = $params['repo']['new_branch'] = drush_get_option('new-branch');
115  }
116  if (drush_get_option('create-branch', FALSE)) {
117  $command_line['create-branch'] = $params['repo']['create_branch'] = drush_get_option('create-branch');
118  }
119 
120  $params['command_line_params'] = $command_line;
121  // PREP AND RETURN PARAMS
122  // Add global locations to the params array to inspect in the --show=params
123  // output.
124  // @backlog Change to allow job name override.
125  $params['global_locations'] = $global_locations;
126  $args = drush_get_arguments();
127  $params['build']['job'] = $args[1];
128  $params['build']['key'] = substr(str_shuffle(md5(microtime())), 0, 5);
129  _rush_derive_parameters($params);
130 
131  return $params;

197  // build job folder name by default plus the env tld.
198  if (empty($params['build']['domain'])) {
199  $args = drush_get_arguments();
200  $job_full_name = $args[1];
201

203  $job_name = str_replace(' ', '_', $job_name);
204  $domain = $job_name . $params['env']['tld'];
205  $derived['build']['domain'] = $params['build']['domain'] = strtolower($domain);
206  }
207 
208  if (empty($params['build']['machine_name'])) {
209  $machine_name = str_replace('.', '_', $params['build']['domain']);
210  $derived['build']['machine_name'] = $params['build']['machine_name'] = $machine_name;
211  }
212 
213  if (empty($params['build']['site_uri'])) {
214  $derived['build']['site_uri'] = $params['build']['site_uri'] = 'http://' . $params['build']['domain'];
215  }
216  // MAMP default port is not set to 80.
217  // Any host port not set to 80 will have the port added to the site url.
218  if ($params['env']['host_port'] != '80') {
219  $derived['build']['site_uri'] = $params['build']['site_uri'] = $params['build']['site_uri'] . ':' . $params['env']['host_port'];
220  }
221 
222  // The build_root_directory_name is usually the name of webroot directory on
223  // the server. Example settings include: 'www', 'websites', and 'Sites'.
224  // If the directory name is not set in the global.ini file then set it to
225  // 'Sites' as a default; which is kind of OSX-centric.
226  if (empty($params['build']['root_directory_name'])) {
227  $derived['build']['root_directory_name'] = $params['build']['root_directory_name'] = 'Sites';
228  }
229  // The build_root_directory_path is usually the path to the webroot directory
230  // on the server. Usually it's something like '~/webroot'.
231  // Note, it does not have to be. A particular Drush Rush job may be just to
232  // download a bunch of code libraries to a user directory for the use of code
233  // completion via an IDE include libraries option. With special cases, the
234  // build_root_directory_path can be changed in a job ini file.
235  if (empty($params['build']['root_directory_path'])) {
236  $derived['build']['root_directory_path'] = $params['build']['root_directory_path'] = $params['global_locations']['RUSH_SERVER_HOME'] . $params['build']['root_directory_name'] . '/';
237  }
238 
239  if (empty($params['build']['backups_directory_name'])) {
240  $derived['build']['backups_directory_name'] = $params['build']['backups_directory_name'] = 'backups';
241  }
242 
243  if (empty($params['build']['backups_directory_path'])) {
244  $derived['build']['backups_directory_path'] = $params['build']['backups_directory_path'] = $derived['build']['root_directory_path'] . $params['build']['backups_directory_name'] . '/';
245  }
246 
247  // If there is no build logs path, then try to find one in the root
248  // directory path.
249  if (empty($params['build']['logs_path'])) {
250  $derived['build']['logs_path'] = $params['build']['logs_path'] = $derived['build']['root_directory_path'] . 'logs/';
251  if (!is_dir($derived['build']['logs_path'])) {
252  // If the directory does not exist, then unset the param.
253  unset($derived['build']['logs_path']);
254  unset($params['build']['logs_path']);
255  }
256  }
257 
258  // The build_directory_name is automatically derived based on the
259  // build_machine_name. However, setting this value in a job params file will
260  // override the name. This is useful if you want a sub job to use the same
261  // path name as the parent job.
262  if (empty($params['build']['directory_name'])) {
263  $derived['build']['directory_name'] = $params['build']['directory_name'] = $params['build']['machine_name'];
264  }
265 
266  // The build_directory_path is the sum of the build_root_directory_path and
267  // the build_directory_name. It's automatically derived. However, setting
268  // this value in a job params file will override the path. In practice, this
269  // would be the only value needed to be overwritten, rather than the

271  if (empty($params['build']['directory_path'])) {
272  $derived['build']['directory_path'] = $params['build']['directory_path'] = $params['build']['root_directory_path'] . $params['build']['directory_name'] . '/';
273  }
274  // A Drupal project stored in version control may have it's docroot
275  // (Drupal codebase) in a sub-directory of the overall project. Drush Rush
276  // must know where the core Drupal code is. Example values include: 'htdocs',

279  $derived['build']['docroot_name'] = $params['build']['docroot_name'] = 'docroot';
280  }
281  // This is the docroot_path used by Drush and Apache Vhosts.
282  if (empty($params['build']['docroot_path'])) {
283  $derived['build']['docroot_path'] = $params['build']['docroot_path'] = $params['build']['directory_path'] . $params['build']['docroot_name'] . '/';
284  }
285  // Build code_directory_path path is typically an alias for the docroot_path.
286  // However, it can be changed if the codebase is located somewhere else.
287  if (empty($params['build']['code_directory_path'])) {
288  $derived['build']['code_directory_path'] = $params['build']['code_directory_path'] = $params['build']['docroot_path'];
289  }
290  // The name of the Drupal site folder, inside the sites directory.
291  if (empty($params['build']['site_folder_name'])) {
292  $derived['build']['site_folder_name'] = $params['build']['site_folder_name'] = 'default';
293  }
294  // Site specification used by Drush.
295  if (empty($params['build']['site_specification'])) {
296  $derived['build']['site_specification'] = $params['build']['site_specification'] = $params['build']['code_directory_path'] . '?uri=' . $params['build']['site_uri'] . '#' . $params['build']['site_folder_name'];
297  }

299  if (empty($params['build']['site_alias_record'])) {
300  $derived['build']['site_alias_record'] = $params['build']['site_alias_record'] = drush_sitealias_get_record($params['build']['site_specification']);
301  }
302  // Site profile used by Drush Site Install.
303  if (empty($params['build']['site_profile'])) {
304  $derived['build']['site_profile'] = $params['build']['site_profile'] = 'standard';
305  }
306

308  if (empty($params['build']['make_file'])) {
309  $derived['build']['make_file'] = $params['build']['make_file'] = 'default.make';
310  }
311

313  // This will auto set a make file if it can find one.
314  if (empty($params['build']['make_file_path'])) {
315  $make_file_path = _rush_derive_make_file_path($params, $params['build']['make_file']);
316  if ($make_file_path != FALSE) {
317  $derived['build']['make_file_path'] = $params['build']['make_file_path'] = $make_file_path;
318  }
319  }
320  // This is not an important parameter, so it's derived by default.
321  // It's used to drop place-holder text into a file when auto-created by
322  // Drush Rush. In keeping with allowing lot's of configuration options, this
323  // setting can also be changed in a global params file or build params file.
324  if (empty($params['build']['file_list_data'])) {
325  $derived['build']['file_list_data'] = $params['build']['file_list_data'] = 'File added by Drush Rush.';
326  }

343 function _rush_derive_parameters_db(&$params, &$derived) {
344 
345  // Default to the db type of mysql.
346  if (empty($params['db']['type'])) {
347  $derived['db']['type'] = $params['db']['type'] = 'mysql';
348  }
349  // Default to the server of localhost.
350  if (empty($params['db']['server'])) {
351  $derived['db']['server'] = $params['db']['server'] = 'localhost';
352  }
353  // Default to the server of 3306.
354  // Note: MAMP runs mysql by default on port: 8889.
355  if (empty($params['db']['port'])) {
356  $derived['db']['port'] = $params['db']['port'] = '3306';
357  }
358  // Default to root. Should be changed in params to a user with just the
359  // appropriate amount of permissions and no more.
360  if (empty($params['db']['user'])) {
361  $derived['db']['user'] = $params['db']['user'] = 'root';
362  }
363  // Default to root. Rarely safe, unless behind an excellent firewall on a
364  // secure box. Change in params as needed.
365  if (empty($params['db']['password'])) {
366  $derived['db']['password'] = $params['db']['password'] = 'root';
367  }
368  // The db name will usually be set in the project params file.
369  // If not, set it to the project machine name.
370  if (empty($params['db']['name'])) {
371  $derived['db']['name'] = $params['db']['name'] = $params['build']['machine_name'];
372  }
373  // The db directory name will usually be set in the project params file.
374  // If not, set it 'sql'.
375  // This is where db files will be stored. SQL dumps or sqlite files.
376  // The directory is assumed to be within the ['build']['directory_path'] but
377  // it doesn't have to be.
378  if (empty($params['db']['directory_name'])) {
379  $derived['db']['directory_name'] = $params['db']['directory_name'] = 'sql';
380  }
381  // Set the default location of the db directory path to be within the build
382  // directory. Change this if the job's db files are located somewhere else
383  // on the system.
384  if (empty($params['db']['directory_path'])) {
385  $derived['db']['directory_path'] = $params['db']['directory_path'] = $params['build']['directory_path'] . $params['db']['directory_name'] . '/';
386  }
387 
388  if (empty($params['db']['file_ext'])) {
389  $derived['db']['file_ext'] = $params['db']['file_ext'] = 'sql';
390  }
391 
392  if (empty($params['db']['file_name'])) {
393  $derived['db']['file_name'] = $params['db']['file_name'] = $params['db']['name'] . '.' . $params['db']['file_ext'];
394  }
395 
396  if (empty($params['db']['file_path'])) {
397  $derived['db']['file_path'] = $params['db']['file_path'] = $params['db']['directory_path'] . $params['db']['file_name'];
398  }
399 
400  // Build the db url based on the database info above.
401  // Override at the project level.
402  if (empty($params['db']['url'])) {
403  // Extract variable names from the array to make the DB URLs easier to read.
404  // extract($params['db']);
405  // Extract scares code inspections, so fo now:
406  $type = $params['db']['type'];
407  $user = $params['db']['user'];
408  $password = $params['db']['password'];
409  $port = $params['db']['port'];
410  $server = $params['db']['server'];
411  $name = $params['db']['name'];
412  $file_path = $params['db']['file_path'];
413  switch ($type) {
414  case 'mysql':
415  $derived['db']['url'] = $params['db']['url'] = "$type://$user:$password@$server:$port/$name";
416  break;
417 
418  case 'sqlite':
419  $derived['db']['url'] = $params['db']['url'] = "$type://$file_path";
420  break;
421  }
422  }
423 }

544  * An array merge function which does not change data types during merge.
545  *
546  * The array_merge_recursive function does merge arrays, but it converts values
547  * with duplicate keys to arrays rather than overwriting the value in the first
548  * array with the duplicate value in the second array, as array_merge does.
549  * I.e., with array_merge_recursive, this happens (documented behavior):
550  *
551  * array_merge_recursive(array('key' => 'org value'),
552  * array('key' => 'new value'));
553  * => array('key' => array('org value', 'new value'));
554  *
555  * _rush_array_merge_recursive_distinct does not change the datatypes of values
556  * in the arrays. Matching keys' values in the second array overwrite those in
557  * the first array, as is the case with array_merge, i.e.:
558  *
559  * _rush_array_merge_recursive_distinct(array('key' => 'org value'),
560  * array('key' => 'new value'));
561  * => array('key' => array('new value'));
562  *
563  * Parameters are passed by reference, though only for performance reasons.
564  * They're not
565  * altered by this function.
566  *
567  * @param array $array1
568  * Array one of two to be merged.
569  * @param array $array2
570  * Array two of two to be merged.
571  *
572  * @return array
573  * An array of merged arrays.
574  */
575 function _rush_array_merge_recursive_distinct(array &$array1, array &$array2) {
576 
577  $merged = $array1;
578 
579  foreach ($array2 as $key => &$value) {
580  if (is_array($value) && isset($merged[$key]) && is_array($merged[$key])) {
581  $merged[$key] = _rush_array_merge_recursive_distinct($merged[$key], $value);
582  }
583  else {
584  $merged[$key] = $value;
585  }
586  }
587 
588  return $merged;
589 }
