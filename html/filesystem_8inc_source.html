45 /**
46  * Extract Rush application folder locations.
47  *
48  * The following locations are derived in this function:
49  * 1. The Rush "application" folder.
50  * - The Rush folder should be in the user's home directory and should be
51  * named "Rush".
52  * - Uses drush_server_home() to find the home directory.
53  * - $global_locations['_RUSH_HOME_DIRECTORY_PATH']
54  * 2. The global environment directory within the Rush application folder.
55  * - This is where global (system environment) params should be placed.
56  * - Using global params allows job folders to inherit system level
57  * information without the need to define the information more than once.
58  * - $global_locations['RUSH_ENVIRONMENT_DIRECTORY_PATH_']
59  * 3. The jobs directory within the Rush application folder.
60  * - This is where single job folders may be placed.
61  * - $global_locations['_RUSH_JOBS_DIRECTORY_PATH_']
62  * 4. The specific job folder defined in the job argument at run time.
63  * - The params and rush files in here define build specific parameters and
64  * operations.
65  * - `drush rush jobOne` will run the params and operations included in
66  * the directory ~/Rush/jobs/jobOne/ .
67  * - $global_locations['_RUSH_JOB_DIRECTORY_PATH_']
68  *
69  * @see rush_drush_command
70  * @see drush_rush_rush
71  * @return array
72  * An associative array with the path locations.
73  */
74 function rush_get_global_locations() {
75 
76  $global_locations['RUSH_HOME_DIRECTORY_NAME'] = 'Rush';
77  $global_locations['RUSH_MODULE_DIRECTORY_PATH'] = rush_self_path();
78  $global_locations['RUSH_ENVIRONMENT_DIRECTORY_NAME'] = 'environment';
79  $global_locations['RUSH_JOBS_DIRECTORY_NAME'] = 'jobs';
80  $global_locations['RUSH_JOB_TEMPLATES_DIRECTORY_NAME'] = 'RushTemplates';
81  $global_locations['RUSH_SERVER_HOME'] = drush_server_home() . '/';
82  $global_locations['RUSH_HOME_DIRECTORY_PATH'] = $global_locations['RUSH_SERVER_HOME'] . $global_locations['RUSH_HOME_DIRECTORY_NAME'] . '/';
83  $global_locations['RUSH_ENVIRONMENT_DIRECTORY_PATH'] = $global_locations['RUSH_HOME_DIRECTORY_PATH'] . $global_locations['RUSH_ENVIRONMENT_DIRECTORY_NAME'] . '/';
84  $global_locations['RUSH_JOBS_DIRECTORY_PATH'] = $global_locations['RUSH_HOME_DIRECTORY_PATH'] . $global_locations['RUSH_JOBS_DIRECTORY_NAME'] . '/';
85  $global_locations['RUSH_JOB_TEMPLATES_DIRECTORY_PATH'] = $global_locations['RUSH_MODULE_DIRECTORY_PATH'] . $global_locations['RUSH_JOB_TEMPLATES_DIRECTORY_NAME'] . '/';
86  $args = drush_get_arguments();
87  $job_arg = $args[1];
88  $job = str_replace('.', "/", $job_arg);
89  $global_locations['RUSH_JOB_DIRECTORY_PATH'] = $global_locations['RUSH_JOBS_DIRECTORY_PATH'] . $job . '/';
90  // Check whether the job directory exists.
91  if (!is_dir($global_locations['RUSH_JOB_DIRECTORY_PATH'])) {
92  // Don't set an error at this time.
93  // drush_rush_validate() will handle error checking.
94  // Instead switch to the job_templates directory.
95  // This is mainly a convenience for Job Template developers to keep template
96  // jobs separate from their own jobs and for people who want to update
97  // template jobs with a simple replacement of the previous directory.
98  $global_locations['RUSH_JOBS_DIRECTORY_PATH'] = $global_locations['RUSH_JOB_TEMPLATES_DIRECTORY_PATH'];
99  $global_locations['RUSH_JOB_DIRECTORY_PATH'] = $global_locations['RUSH_JOBS_DIRECTORY_PATH'] . $job . '/';
100  }
101  // Check whether this is a sub job and if so, identify the top job directory.
102  $job_directories = explode('.', $job_arg);
103  if (count($job_directories) > 1) {
104  $global_locations['RUSH_JOB_TOP_DIRECTORY_PATH'] = $global_locations['RUSH_JOBS_DIRECTORY_PATH'] . $job_directories[0] . '/';
105  }
106 
107  return $global_locations;
108 }

110 /**
111  * List the directories and files in the specified root directory.
112  *
113  * @param string $root
114  * Required. The root directory to begin listing within.
115  *
116  * @return array
117  * An array of directories and files.
118  * $files['files'] is a list of all files.
119  * $files['dirs'] is a list of all directories.
120  */
121 function rush_list_files($root = '.') {
122 
123  $files = array('files' => array(), 'dirs' => array());
124  $directories = array();
125  $last_letter = $root[strlen($root) - 1];
126  $root = ($last_letter == '\\' || $last_letter == '/') ? $root : $root . DIRECTORY_SEPARATOR;
127 
128  $directories[] = $root;
129 
130  while (count($directories)) {
131  $dir = array_pop($directories);
132  $handle = opendir($dir);
133  if ($handle) {
134  while (FALSE !== ($file = readdir($handle))) {
135  if ($file == '.' || $file == '..' || $file == '.git') {
136  continue;
137  }
138  $file = $dir . $file;
139  if (is_dir($file)) {
140  $directory_path = $file . DIRECTORY_SEPARATOR;
141  array_push($directories, $directory_path);
142  $files['dirs'][] = $directory_path;
143  }
144  elseif (is_file($file)) {
145  $files['files'][] = $file;
146  }
147  }
148  closedir($handle);
149  }
150  }
151 
152  return $files;
153 }

168 function _rush_derive_make_file_path($params, $make_file = 'default.make') {
169 
170  // Look for the make file in the build directory.
171  // This first directory will not exist if it is a fresh build. However, this
172  // function will be called again when drush make is called, so a make file in
173  // a build directory can be used.
174  $make_file_path = $params['build']['directory_path'] . $make_file;
175  if (file_exists($make_file_path)) {
176  return $make_file_path;
177  }
178 
179  // Look for the make file in the job directory.
180  $make_file_path = $params['global_locations']['RUSH_JOB_DIRECTORY_PATH'] . $make_file;
181  if (file_exists($make_file_path)) {
182  return $make_file_path;
183  }
184  // Look for the make file in the top job directory.
185  $make_file_path = $params['global_locations']['RUSH_JOB_TOP_DIRECTORY_PATH'] . $make_file;
186  if (file_exists($make_file_path)) {
187  return $make_file_path;
188  }
189  // Look for the make file in the Rush global directory.
190  $make_file_path = $params['global_locations']['RUSH_ENVIRONMENT_DIRECTORY_PATH'] . $make_file;
191  if (file_exists($make_file_path)) {
192  return $make_file_path;
193  }
194 
195  return FALSE;
196 }
