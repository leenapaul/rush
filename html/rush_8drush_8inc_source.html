209  * Run a rush file with commands in it.
210  *
211  * @see drush_invoke()
212  * @see drush.api.php

214  * @backlog Refactor switch case statements.
215  * @return bool
216  * FALSE on failure.

218 function drush_rush() {
219 
220  $params = rush_get_params();
221  $rush_operations = rush_get_operations($params);
222  if (drush_get_option('create-branch', FALSE)) {
223  $rush_operations[]['f'] = 'git_create_branch';
224 
225  }
226  if (drush_get_option('show', FALSE)) {
227  $show_mode = drush_get_option('show');
228 
229  return _rush_show($params, $show_mode, $rush_operations);
230  }
231 
232 
233  $msg = dt('[ERROR]: An error occurred while running this drush operation: ');
234  // @backlog Make a progress meter based on number of operations to run.
235  // @backlog Have a finished message.
236  foreach ($rush_operations as $op_array) {
237  foreach ($op_array as $op_type => $op) {
238  switch ($op_type) {
239  // The op_type is a function.
240  case 'f':
241  if (rush_function_exec($op, $params) === FALSE) {
242  return drush_set_error('DRUSH_RUSH_OP_ERROR', $msg . $op);
243  }
244  break;
245 
246  // The op_type is a command.
247  case 'c':
248  if (rush_command_exec($op, $params) === FALSE) {
249  return drush_set_error('DRUSH_RUSH_OP_ERROR', $msg . $op);
250  }
251  break;
252 
253  // The op_type is a job to run.
254  case 'j':
255  if (rush_job_exec($op, $params) === FALSE) {
256  return drush_set_error('DRUSH_RUSH_OP_ERROR', $msg . $op);
257  }
258  break;
259 
260  // The op_type is a message.
261  case 'm':
262  if (rush_message_exec($op, $params) === FALSE) {
263  return drush_set_error('DRUSH_RUSH_OP_ERROR', $msg . $op);
264  }
265  break;
266  }
267  }

295  switch ($show_mode) {

297  case 'jt':
298  $files = rush_list_files($global_locations['RUSH_JOB_TEMPLATES_DIRECTORY_PATH']);
299  $msg = rush_message_wrapper(dt('JOB TEMPLATES LIST'), 0);

302  break;
303 
304  case 'locations':
305  case 'l':
306  $msg = rush_message_wrapper(dt('GLOBAL LOCATIONS'), 0);

309  drush_print_r($global_locations);
310 
311  return TRUE;
312 
313  default:

315 
316  return drush_set_error('DRUSH_RUSH_INVALID_OPTION', $msg);
317  }
318  }
319 
320  // Tell the user what list they will get.
321  drush_print($msg);
322  $jobs_list = $files['dirs'];
323  // Sort the directory names by alpha order.
324  natcasesort($jobs_list);
325 
326  foreach ($jobs_list as $job) {
327  $job = explode("/$rush_dir_name/$directory_name/", $job);
328  $job = trim($job[1], '/');
329  $job = str_replace('/', '.', $job);

333  drush_print($msg);

383 function rush_command_exec($command, $params = array()) {
384 
385  // If this is a drush command then run it in the docroot directory.
386  $pos = strpos($command, 'drush');
387  // Using != would not work as expected
388  // because a position can be 0. The statement (0 != false) evaluates
389  // to false.
390  if ($pos !== FALSE) {
391  $command = 'cd ' . $params['build']['docroot_path'] . '; ' . $command . ' --uri=' . $params['build']['site_uri'];
392  }
393  // If it's not a drush command, then run it in the build directory.
394  else {
395  $command = 'cd ' . $params['build']['directory_path'] . '; ' . $command;
396  }
397  if (drush_shell_exec($command)) {
398  $msg = dt('[RUSH COMMAND EXEC]: !command', array('!command' => $command));
399  drush_log($msg, 'success');
400 
401  return TRUE;
402  }
403  else {
404  $msg = dt('!command', array('!command' => $command));
405 
406  return drush_set_error('RUSH_COMMAND_EXEC_FAILED', $msg);
407 
408  }
409 }
