56 function rush_invoke($alias, $command, $args, $options, $function, $type = 'message') {
57 
58  $cmd_string = $command . ' ' . implode(' ', $args) . ' ' . implode(' ', $options);
59  if (drush_invoke_process($alias, $command, $args, $options, TRUE)) {

61  $msg = dt('[RUN DRUSH]: !cmd_string', array('!cmd_string' => $cmd_string));
62  drush_log($msg, $type);
63 
64  return TRUE;

67 
68  $msg = dt('[ERROR]: *** !operation failed to run: !cmd_string ***', array(
69  '!cmd_string' => $cmd_string,

73  return drush_set_error('RUSH_INVOKE_ERROR', $msg);
